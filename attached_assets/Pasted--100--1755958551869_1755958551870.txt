المتطلبات

# تثبيت الأدوات

npm install -g @expo/cli eas-cli

البناء المحلي

# الانتقال للمجلد

cd mobile-app

# إعداد متغيرات البيئة

cp .env.example .env

# تعديل .env وإضافة بيانات Supabase

# بدء التطوير

npx expo start

بناء APK

# بناء APK للاختبار

eas build --platform android --profile preview

# بناء للإنتاج

eas build --platform android --profile production
